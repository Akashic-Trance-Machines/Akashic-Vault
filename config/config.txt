# config.txt — Akashic Vault (Circle bare-metal)
#
# One SD card serves both boards: the Pi3 loads kernel8.img and the Pi4 loads
# kernel8-rpi4.img automatically. Circle does not use an armstub on Pi3.

arm_64bit=1
kernel_address=0x80000
kernel=kernel8.img

# Keep the GPU/CPU split small — we use I2S, not HDMI audio.
gpu_mem=64

# I2S audio out (DAC on the HAT). Circle drives I2S directly.
dtparam=i2s=on

# Disable Bluetooth and explicitly enable PL011 UART on GPIO14/15 for TRS MIDI.
dtoverlay=disable-bt
enable_uart=1

# Pi4-specific overrides (mirrors Circle's reference boot/config64.txt).
# - kernel= must be set here: the global kernel8.img would otherwise be
#   inherited and the Pi3 build loaded (wrong peripheral base → all silent).
# - armstub8-rpi4.bin is the Circle-specific ARM stub (GIC setup); built from
#   circle/boot/armstub, shipped in config/. Required to start Circle on Pi4.
# - bcm2711-rpi-4-b.dtb must also be on the card (deploy script copies it).
[pi4]
arm_boost=1
armstub=armstub8-rpi4.bin
kernel=kernel8-rpi4.img
max_framebuffers=2
[all]
